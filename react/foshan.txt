数据中心: models\index.js     （类似vuex）
1.调用接口：import * as services from '../services'          services.query
export default {
    namespace: '',// 与index的class名一致（第一个字母小写）
    state: {// 类似vuex的state
        product: {
            list: [],
            pagination: defaultPagination
        }
    },
    reducers: {// 类似vuex的mutation
        save(state, { payload }) {// 参数：state，{参数}
            return { ...state, product: payload };// 保存到state，注意写法
        }
    },
    effects: {// 类似vuex的action
        *fetch({ payload = {} }, { call, put, select }) {// 参数：{参数}该写法参数空时为对象，{ call, put, select }
	// 声明时需要添加*，普通函数内部不能使用yield关键字，否则会出错
	// ES6 yield       yield暂停，next()继续
	// React dva 的使用
	// call调用 action  或者 call 调用 async的action函数
	// put调用reducers的函数来更新state
        }
    }
}
